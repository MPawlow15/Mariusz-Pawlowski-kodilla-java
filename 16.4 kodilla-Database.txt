SELECT ISSUES.ISSUESLIST_ID, ISSUES.SUMMARY, ISSUES.DESCRIPTION, ISSUES.USER_ID_ASSIGNEDTO, ISSUESLISTS.NAME
FROM ISSUES
JOIN ISSUESLISTS on ISSUES.ISSUESLIST_ID = ISSUESLISTS.ID;

SELECT ISSUES.ISSUESLIST_ID, ISSUES.SUMMARY, ISSUES.DESCRIPTION, ISSUES.USER_ID_ASSIGNEDTO, USERS.FIRSTNAME, USERS.LASTNAME
FROM ISSUES
JOIN USERS on ISSUES.USER_ID_ASSIGNEDTO = USERS.ID;

SELECT FIRSTNAME, LASTNAME, COUNT(*) AS NUMBER_OF_TASKS
FROM USERS
GROUP BY FIRSTNAME, LASTNAME;
